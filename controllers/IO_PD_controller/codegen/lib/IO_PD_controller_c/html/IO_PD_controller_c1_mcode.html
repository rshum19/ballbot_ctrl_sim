1function [u,uparts] = IO_PD_controller_c(x,x_d)
 2% IO_PD_controller  I/O feedback linearized controller with an outer loop PD
 3% AUTHOR:   Roberto Shu
 4% LAST EDIT: 8/26/2017    
 5    %% ----------------------------------------------------------
 6    %   Unpack actual and desired state vectors
 7    % -----------------------------------------------------------    
 8    % State variables
 9    theta = x(1);
10    phi = x(2);
11    dtheta = x(3);
12    dphi = x(4);
13    
14    % Desired state
15    theta_d = x_d(1);
16    phi_d = x_d(2);
17    dtheta_d = x_d(3);
18    dphi_d = x_d(4);
19    
20    %% ----------------------------------------------------------
21    %   Outer loop control
22    % ----------------------------------------------------------- 
23    % Aim: get a desired lean angle (phi_des) to track desired ball
24    % position using a simple PD controller
25    
26    % Error state vector
27	err = [theta-theta_d; dtheta - dtheta_d;...
28		phi-phi_d; dphi-dphi_d];
29	
30    % Control gains 
31    K_ilc = [0.01 0.015 0 0];
32    
33    % Augmented desired lean angle
34	phi_des = phi_d + K_ilc*err;
35    
36    %% ----------------------------------------------------------
37    %   Inner loop control
38    % ----------------------------------------------------------- 
39    % Aim: I/O feedback linearized controller with PD to track desired lean angles
40    % to balance and navigate
41    
42    % Output function
43    y = phi - phi_des;
44    dy = dphi - dphi_d;
45    y_vec = [y;dy];
46    
47    % Output function lie derative 
48    [LgLfh, Lf2h] = autofun_LieDerv_ballbot2D_IO_PD(theta,phi,dtheta,dphi,phi_des,dphi_d);
49
50    % Control gains 
51    K = [7.0711    4.9135];
52    K = [22.3607   12.0300];
53    K = [50     15];
54    
55    % Feedfoward term
56    u_ff = -(LgLfh)^(-1)*Lf2h;
57    
58    % PD term
59    mu = -K*y_vec;
60    u_fb = (LgLfh)^(-1)*mu;
61    
62    % Full I/O controller
63    u = u_ff + u_fb;
64    
65    % Bound input
66    u_max = 20;
67	if(abs(u)>u_max)
68 		u = sign(u)*u_max;
69	end
70    %% ----------------------------------------------------------
71    %   Pack controller relevant info
72    % ----------------------------------------------------------- 
73    uparts.u_ff = u_ff;
74    uparts.u_fb = u_fb;
75    uparts.mu = mu;
76    uparts.phi_des = phi_des;
77    uparts.kerr = K_ilc*err;
78    uparts.phi_d = phi_d;
79    
80    uparts.V_eps = 0;
81    uparts.d1 = 0;
82    uparts.psi0 =0;
83    uparts.psi1 = 0;
84end
